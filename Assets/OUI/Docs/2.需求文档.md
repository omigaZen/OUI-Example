# OUI UI管理系统需求文档

目标：基于UGUI 简单易用的UI管理系统

- 依赖：UniTask

- 以Window为单位进行层级管理 先按照window的windowLayer排序 相同WindowLayer的后打开（OUI.OpenAsync）的显示到前面 window打开时 如果标记了HideBelow 要隐藏所有层级更小的window以节省性能 具体实现参考：E:\Projects\ManaPaws\Main\UI
- 约束：一个Window类只能有一个实例 如果重复打开一个Window 发现Window已经打开了就直接return
- 再Window类上面加WindowAttribute来标识Window对应的UI资源、Layer、HideBelow信息

```C#
/// <summary>
    /// UI行为接口。
    /// </summary>
    public abstract class BaseWindow: MonoBehaviour
    {
        /// <summary>
        /// window资源加载完成后 Init之前调用
        /// </summary>
        public abstract void ScriptGenerator();

        /// <summary>
        /// ScriptGenerator()之后调用
        /// </summary>
        public abstract void Init();

        /// <summary>
        /// 释放窗口资源时调用
        /// </summary>
        public abstract void Release();
        
        /// <summary>
        /// 打开窗口时调用
        /// </summary>
        public virtual void OnOpen(params object[] args)
        {
            gameObject.SetActive(true);
        }

        /// <summary>
        /// 关闭窗口时调用
        /// </summary>
        public virtual void OnClose()
        {
            gameObject.SetActive(false);
        }

        private bool _hideByAbove = false;
        public bool HideByAbove
        {
            get => _hideByAbove;
            set
            {
                if (_hideByAbove == value) return;
                // todo 被屏蔽逻辑
                if (value)
                {
                    OnHideByAbove();
                }
                else
                {
                    OnShowByAbove();
                }

                _hideByAbove = value;
            }
        }

        protected virtual void OnHideByAbove()
        {
        }

        protected virtual void OnShowByAbove()
        {
        }
    }
```

WindowAttribute：
```C#
/// <summary>
    /// UI层级枚举。
    /// </summary>
    public enum WindowLayer : int
    {
        Bottom = 0,
        UI = 1,
        Top = 2,
        Tips = 3,
        System = 4,
    }

    /// <summary>
    /// 窗口特性类。
    /// </summary>
    [AttributeUsage(AttributeTargets.Class)]
    public class WindowAttribute : Attribute
    {
        /// <summary>
        /// 窗口层级
        /// </summary>
        public readonly WindowLayer WindowLayer;

        /// <summary>
        /// 资源定位地址。
        /// </summary>
        public readonly string Location;

        /// <summary>
        /// 打开窗口后 屏蔽底下的所有window
        /// </summary>
        public readonly bool HideBelow;

        public WindowAttribute(WindowLayer windowLayer, string location = "", bool hideBelow = false)
        {
            WindowLayer = windowLayer;
            Location = location;
            HideBelow = hideBelow;
        }
    }
```


## 编辑器工具

### 代码生成
- Unity编辑器 hierachy中节点右键菜单点击“生成OUI代码”生成代码到粘贴板

- 生成代码示例：
    - 绑定对象：子transform中名称带有“m_{类型标识}”的 
    - 路径：相对于当前节点的路径
```C#

        #region 脚本工具生成的代码

        private Image m_img_bg_up;
        private Transform m_tf_hero;
        private Image m_img_light;
        private Button m_btn_continue;
        private GameObject m_item_skill;
        private GameObject m_item_skill_1;
        private GameObject m_go_got_new;
        private TextMeshProUGUI m_tmp;
        private Transform m_tf_q;
        private TextMeshProUGUI m_tmp_nickname;
        private GameObject m_go_ele;
        private GameObject m_go_job;
        private TextMeshProUGUI m_tmp_name;
        private Image m_img_q;
        private TextMeshProUGUI m_tmp_q;

        public override void ScriptGenerator()
        {
            m_img_bg_up = FindChildComponent<Image>("m_img_bg_up");
            m_tf_hero = FindChild("m_tf_hero");
            m_img_light = FindChildComponent<Image>("m_tf_hero/m_img_light");
            m_btn_continue = FindChildComponent<Button>("m_btn_continue");
            m_item_skill = FindChild("down/m_item_skill").gameObject;
            m_item_skill_1 = FindChild("down/m_item_skill_1").gameObject;
            m_go_got_new = FindChild("down/m_go_got_new").gameObject;
            m_tmp = FindChildComponent<TextMeshProUGUI>("down/m_go_got_new/m_tmp");
            m_tf_q = FindChild("m_tf_q");
            m_tmp_nickname = FindChildComponent<TextMeshProUGUI>("m_tf_q/hlay/m_tmp_nickname");
            m_go_ele = FindChild("m_tf_q/hlay/Image_1/m_go_ele").gameObject;
            m_go_job = FindChild("m_tf_q/hlay/Image_2/m_go_job").gameObject;
            m_tmp_name = FindChildComponent<TextMeshProUGUI>("m_tf_q/m_tmp_name");
            m_img_q = FindChildComponent<Image>("m_tf_q/m_img_q");
            m_tmp_q = FindChildComponent<TextMeshProUGUI>("m_tf_q/m_img_q/m_tmp_q");
            m_btn_continue.onClick.AddListener(OnClick_continueBtn);
        }

        #endregion
```
